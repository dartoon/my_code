# IMAGE and GALFIT CONTROL PARAMETERS
A) var_datafits            # Input data image (FITS file)
B) imgblock.fits       # Output data image block
C) var_noisefits                # Sigma image name (made from data if blank or "none") 
D) var_psffits   #        # Input PSF image and (optional) diffusion kernel
E) 1                   # PSF fine sampling factor relative to data 
F) none                # Bad pixel mask (FITS image or ASCII coord list)
G) none                # File with parameter constraints (ASCII file) 
H) 1    var_size   1    var_size   # Image region to fit (xmin xmax ymin ymax)
I) var_conv_size    var_conv_size          # Size of the convolution box (x y)
J) var_zp              # Magnitude photometric zeropoint 
K) var_resolution var_resolution     # Plate scale (dx dy)    [arcsec per pixel]
O) regular             # Display type (regular, curses, both)
P) 0                   # Choose: 0=optimize, 1=model, 2=imgblock, 3=subcomps
